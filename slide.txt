% Slide 1: Motivation
\begin{frame}{Context and Motivation\\\rule{\linewidth}{0.5pt}} \label{context}
\small
\bsq \textbf{Global Fraud Losses}: Rapid digital transaction growth drives billions in annual credit card fraud losses.\\
\bsq \textbf{Static Rules Failure}: Traditional engines struggle with concept drift and complex, non-linear relationships.\\
\bsq \textbf{Isolated Check Limit}: Single-transaction checks ignore sequential context, leading to high false negatives.\\
\bsq \textbf{Core Goal}: Propose a deep sequential Transformer classifier integrated with a low-latency Spring Boot backend.
\end{frame} 

% Slide 2: Objectives & Scope
\begin{frame}{Project Objectives and Scope\\\rule{\linewidth}{0.5pt}} \label{objectives}
\small
\bsq \textbf{Sequence Modeling}: Optimize a Transformer classifier using transaction hour, category, and log-amount.\\
\bsq \textbf{Feature Engineering}: Extract context features, normalize amounts, and aggregate database-driven user profiles.\\
\bsq \textbf{Imbalance Mitigation}: Apply Random Under-Sampling (RUS) to establish a balanced 1:1 training set.\\
\bsq \textbf{Micro-Latency Execution}: Export Python models to ONNX and embed in-memory inside Spring Boot.\\
\bsq \textbf{Scope}: Evaluated on the synthetic Sparkov dataset spanning 1.8M transactions (Jan 2019 – Dec 2020).
\end{frame}

% Slide 3: Behavioral Feature Engineering
\begin{frame}{Behavioral Feature Engineering Pipeline\\\rule{\linewidth}{0.5pt}} \label{preprocessing}
\small
\bsq \textbf{Feature Extraction}: Extract transaction execution hour, merchant category, and raw amount directly.\\
\bsq \textbf{Feature Aggregation}: Query historical user stats (fraud rate, frequency, success rate, verification) to feed risk engines.\\
\bsq \textbf{Log Normalization}: Apply logarithmic scaling to continuous numerical attributes to compress dynamic range:
\begin{equation}
x_{\text{amt}} = \ln(\text{amount} + 1)
\end{equation}
\end{frame}

% Slide 4: Imbalance Mitigation & Input Feature Space
\begin{frame}{Imbalance Mitigation \& Input Feature Space\\\rule{\linewidth}{0.5pt}} \label{imbalance_features}
\small
\bsq \textbf{Imbalance Mitigation}: Downsample majority class to a 1:1 training ratio via Random Under-Sampling (RUS).\\
\bsq \textbf{Model Input Triplet}: Processes three context features for each transaction sequence:
\begin{itemize}
    \item \textbf{Hour of Day}: Captures diurnal behavioral patterns (e.g., night checkout spikes).
    \item \textbf{Merchant Category}: Identifies the purchase domain (e.g., shopping, travel, utilities).
    \item \textbf{Log-Amount ($x_{\text{amt}}$)}: Represents log-scaled transaction size to stabilize gradient updates.
\end{itemize}\\
\bsq \textbf{Prediction Goal}: Map sequence-level dependencies across these token inputs to calculate transaction fraud risk.
\end{frame}

% Slide 5: Transformer Architecture & Math Foundations
\begin{frame}{Transformer Architecture \& Math Foundations\\\rule{\linewidth}{0.5pt}} \label{transformer_architecture}
\begin{columns}
\begin{column}{0.42\textwidth}
\small
\bsq \textbf{Input Serialization}:
\begin{equation*}
\mathbf{X}_{\text{in}} = \text{Concat}(\text{Hr}, \text{Cat}, x_{\text{amt}})
\end{equation*}
maps inputs to $\mathbb{R}^{3 \times d_{\text{model}}}$.\\
\bsq \textbf{Positional Sum}: Adds sinusoidal coordinates to preserve feature identities.\\
\bsq \textbf{Self-Attention Core}: Maps cross-feature dependencies via projections:
\begin{equation*}
\text{Attn} = \text{softmax}\left(\frac{\mathbf{Q}\mathbf{K}^T}{\sqrt{d_k}}\right)\mathbf{V}
\end{equation*}
\end{column}

% Slide 6: Risk & Trust Scoring Engines
\begin{frame}{Risk \& Trust Scoring Engines\\\rule{\linewidth}{0.5pt}} \label{behavioral_intel}
\small
\bsq \textbf{Risk Engine}: Aggregates model predictions, night checkout rates, and frequency into a Risk Score $\mathcal{R} \in [0, 100]$:
\begin{equation*}
\mathcal{R} = 0.40 F_r + 0.30 P_{avg} + 0.20 N_r + 0.10 \min(N_{\text{transaction}}, 100)
\end{equation*}\\
\bsq \textbf{Trust Engine}: Tracks long-term reputation via account age, successful checkouts, and verification status $\mathcal{T} \in [0, 100]$:
\begin{equation*}
\mathcal{T} = 0.20 A + 0.30 S + 0.30 H + 0.10 V + 0.10 P
\end{equation*}
\end{frame}

% Slide 11: Limitations and Future Work
\begin{frame}{Limitations and Future Work\\\rule{\linewidth}{0.5pt}} \label{limitations}
\small
\bsq \textbf{Data Attrition}: Random Under-Sampling discards over 90% of genuine samples, limiting clean behavior diversity.\\
\bsq \textbf{Weight Staleness}: Embedded ONNX models do not update weights dynamically in response to online feedback.\\
\bsq \textbf{SMOTE \& Focal Loss (Roadmap)}: Preserve legitimate samples via oversampling, and shift to skew-sensitive Focal Loss:
\begin{equation*}
\mathcal{L}_{\text{FL}}(p_t) = -\alpha_t (1 - p_t)^\gamma \log(p_t)
\end{equation*}\\
\bsq \textbf{Input Context Extension (Roadmap)}: Embed dynamic, long-term database-aggregated user profile metrics directly into the Transformer.
\end{frame}
